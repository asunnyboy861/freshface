```swift
// ProductSortOption.swift (新增)
enum ProductSortOption: String, CaseIterable {
    case name = "Name"
    case dateAdded = "Date Added"
    case expiryDate = "Expiry Date"
    case riskLevel = "Risk Level"

    var icon: String {
        switch self {
        case .name: return "textformat.abc"
        case .dateAdded: return "calendar"
        case .expiryDate: return "clock"
        case .riskLevel: return "exclamationmark.triangle"
        }
    }
}

// ProductListViewModel 扩展方法
extension ProductListViewModel {
    func sortProducts() {
        switch sortOption {
        case .name:
            filteredProducts.sort { $0.name < $1.name }
        case .dateAdded:
            filteredProducts.sort { $0.createdAt > $1.createdAt }
        case .expiryDate:
            filteredProducts.sort { ($0.daysRemaining ?? Int.max) < ($1.daysRemaining ?? Int.max) }
        case .riskLevel:
            filteredProducts.sort { ($0.riskLevel?.sortOrder ?? 99) < ($1.riskLevel?.sortOrder ?? 99) }
        }
    }

    func applyFilterAndSort() {
        // Apply category filter first
        filterByCategory(currentFilterCategory)

        // Then apply sort
        sortProducts()
    }
}

extension RiskLevel {
    var sortOrder: Int {
        switch self {
        case .expired: return 0
        case .critical: return 1
        case .warning: return 2
        case .caution: return 3
        case .safe: return 4
        }
    }
}
```

## 7. 设计原则与约束

### 7.1 功能模块原则

```
✅ 单一职责: 每个文件只负责一个功能
✅ 高内聚: 相关功能放同一模块
✅ 低耦合: 模块间通过协议通信
✅ 命名清晰: 文件名 = 功能名
✅ 语义化: 目录结构反映业务结构
```

### 7.2 代码复用原则 (三次法则)

```
第1次出现: 直接写
第2次出现: 复制粘贴 (可接受)
第3次出现: 抽象提取为通用组件/方法

✅ 已复用:
- ProductCategory 枚举 (M2 不再重复创建)
- CloudProductRepository (已有 CRUD)
- NotificationManager (已有)
- CloudKitManager (已有)

🔄 待提取:
- PolicySection (M4 PrivacyPolicyView 中重复)
```

### 7.3 废弃代码清理流程

```
1. 标记废弃: @available(*, deprecated, message: "...")
2. 验证无影响: 确认无调用点
3. 删除代码: 移除废弃代码
4. 提交信息: "refactor: remove deprecated XXX"
```

### 7.4 继承与兼容性原则

