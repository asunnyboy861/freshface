### 1.1 背景
基于对FreshFace软件的全面分析，发现以下需要优化的问题：

1. 修复HomeView空导航函数，实现完整的用户引导流程 ✅

- **风格一致**: 与现有代码风格、UI风格保持一致

### 2.2 Repository使用策略统一

#### 需求描述
当前项目中Repository使用不一致：
- HomeView使用`LocalProductRepository()`
- ProductListView使用`LocalProductRepository()`
- AnalyticsView使用`LocalProductRepository()`
- RoutineDetailView使用`CloudProductRepository()`

这导致数据可能不同步，需要统一策略。

#### 问题分析
```swift
// 当前代码 - 多处创建新实例
@StateObject private var viewModel = ProductListViewModel(repository: LocalProductRepository())

// RoutineDetailView.swift
private let repository = CloudRoutineRepository()
```

#### 实际实现方案 ✅
采用**依赖注入 + EnvironmentObject**模式，通过AppCoordinator统一管理Repository实例，**与SettingsView的iCloud设置通过UserDefaults同步**：

```swift
// ============================================
// AppCoordinator.swift - 实际实现
// 位置: Presentation/Coordinators/AppCoordinator.swift
// 实现说明: 使用UserDefaults与Settings的iCloud设置保持同步
// ============================================

class AppCoordinator: ObservableObject {
    @Published var selectedTab: AppTab = .home
    @Published var selectedProduct: Product?
    @Published var showAddProduct = false
    @Published var showScanner = false

    private lazy var localProductRepository: ProductRepositoryProtocol = LocalProductRepository()
    private lazy var cloudProductRepository: ProductRepositoryProtocol = CloudProductRepository()
    private lazy var localRoutineRepository: RoutineRepositoryProtocol = LocalRoutineRepository()
    private lazy var cloudRoutineRepository: RoutineRepositoryProtocol = CloudRoutineRepository()

    // [KEY] 使用UserDefaults与Settings的@AppStorage("iCloudSyncEnabled")保持同步
    var productRepository: ProductRepositoryProtocol {
        if UserDefaults.standard.bool(forKey: "iCloudSyncEnabled") {
            return cloudProductRepository
        }
        return localProductRepository
    }

    var routineRepository: RoutineRepositoryProtocol {
        if UserDefaults.standard.bool(forKey: "iCloudSyncEnabled") {
            return cloudRoutineRepository
        }
        return localRoutineRepository
    }

    // ... 其他代码 ...
}
```

#### 关键设计决策
1. **与Settings同步**: AppCoordinator通过`UserDefaults.standard.bool(forKey: "iCloudSyncEnabled")`读取iCloud设置，与SettingsView的`@AppStorage("iCloudSyncEnabled")`保持同步
2. **懒加载单例**: 各Repository实例使用`lazy var`懒加载，确保单例行为
3. **computed property**: `productRepository`和`routineRepository`使用计算属性，每次访问实时读取设置

#### 检验标准 ✅
- [x] AppCoordinator的productRepository/routineRepository与SettingsView的iCloud设置同步
- [x] 切换iCloud同步设置后，AppCoordinator返回正确的Repository类型
- [x] 所有View通过coordinator.productRepository访问数据，保证一致性

#### 测试用例

import SwiftUI

struct ProductDetailView: View {

        ))

        ScrollView {
            VStack(spacing: 20) {
                productImageView
                productInfoSection
                expiryStatusSection

                usageSection
                actionsSection
            }
            .padding()

            ShareSheet(activityItems: shareItems)
        }
    }
    
    // ... 其他代码保持不变 ...
    
    // [MODIFY] 实现分享功能
    private func shareProduct() {
        // 构建分享内容
        let product = viewModel.product
        var shareText = "🧴 \(product.name)"
        
        if let brand = product.brand {
            shareText += " by \(brand)"
        }
        
        shareText += "\n📦 Category: \(product.category.displayName)"
        
        if let type = product.productType {
            shareText += "\n🏷️ Type: \(type.displayName)"
        }
        
        // 添加过期状态
        if let risk = viewModel.riskLevel, let days = viewModel.daysRemaining {
            let statusEmoji = days <= 0 ? "❌" : days <= 7 ? "⚠️" : "✅"
            shareText += "\n\(statusEmoji) Status: \(risk.rawValue)"
            
            if days > 0 {
                shareText += " (\(days) days remaining)"
            } else {
                shareText += " (Expired)"
            }
        }
        
        // 添加FreshFace品牌
        shareText += "\n\n📱 Tracked with FreshFace"
        
        // 准备分享项
        var items: [Any] = [shareText]
        
        // 如果有产品图片，添加到分享项
        if let imageData = product.imageData, let image = UIImage(data: imageData) {
            items.append(image)
        }
        
        shareItems = items
        showShareSheet = true
    }
}

import SwiftUI

struct ProductRecommendationsView: View {
    let category: ProductCategory
    let currentProduct: Product?
    
    // [MODIFY] 使用EnvironmentObject获取coordinator
    @EnvironmentObject private var coordinator: AppCoordinator
    @StateObject private var viewModel = ProductRecommendationsViewModel()
    
    // [NEW] 添加状态
    @State private var showAddSuccessMessage = false
    @State private var addedProductName = ""
    @State private var showProductDetail = false
    @State private var selectedRecommendation: ProductRecommendation?
    
    var body: some View {
        ScrollView {
            VStack(spacing: 20) {
                headerSection
                
                if viewModel.isLoading {
                    ProgressView("Finding clean alternatives...")
                        .padding()
                } else if viewModel.recommendations.isEmpty {
                    emptyStateView
                } else {
                    recommendationsList
                }
            }
            .padding()
        }
        .navigationTitle("Clean Alternatives")
        .navigationBarTitleDisplayMode(.large)
        .onAppear {
            viewModel.loadRecommendations(
                for: category, 
                excluding: currentProduct?.id,
                repository: coordinator.productRepository
            )
        }
        // [NEW] 添加成功提示
        .alert("Added to Library", isPresented: $showAddSuccessMessage) {
            Button("OK") { }
            Button("View Products") {
                coordinator.navigate(to: .products)
            }
        } message: {
            Text("\(addedProductName) has been added to your product library.")
        }
        // [NEW] 产品详情sheet
        .sheet(item: $selectedRecommendation) { recommendation in
            NavigationStack {
                RecommendedProductDetailView(recommendation: recommendation)
                    .toolbar {
                        ToolbarItem(placement: .topBarTrailing) {
                            Button("Done") {
                                selectedRecommendation = nil
                            }
                        }
                    }
            }
        }
    }
    
    // ... headerSection和emptyStateView保持不变 ...
    
    private var recommendationsList: some View {
        LazyVStack(spacing: 16) {
            ForEach(viewModel.recommendations) { recommendation in
                CleanAlternativeCard(
                    recommendation: recommendation,
                    onAddToLibrary: {
                        addToLibrary(recommendation)
                    },
                    onViewDetails: {
                        selectedRecommendation = recommendation
                    }
                )
            }
        }
    }
    
    // [NEW] 添加到产品库
    private func addToLibrary(_ recommendation: ProductRecommendation) {
        Task {
            do {
                // 创建新产品（复制推荐产品的信息）
                let newProduct = Product(
                    name: recommendation.product.name,
                    brand: recommendation.product.brand,
                    category: recommendation.product.category,
                    productType: recommendation.product.productType,
                    imageData: recommendation.product.imageData,
                    ingredients: recommendation.product.ingredients
                )
                
                // 保存到Repository
                _ = try await coordinator.productRepository.create(newProduct)
                
                // 显示成功消息
                await MainActor.run {
                    addedProductName = newProduct.name
                    showAddSuccessMessage = true
                }
            } catch {
                print("Failed to add product: \(error)")
            }
        }
    }
}
