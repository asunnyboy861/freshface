## 修订说明

### 背景

基于 V3.1 实施完成后的代码审查，发现以下问题需要优化：

| 问题编号 | 问题描述 | 影响 | 优先级 |
|---------|---------|------|--------|
| **S1** | RoutineStepEditorView availableProducts 传入空数组 | 用户无法选择已有产品 | P0 |
| **S2** | Barcode 扫描后无法自动填充产品信息 | 条码功能不完整 | P0 |
| **S3** | HomeView 返回后数据不刷新 | 用户看不到最新数据 | P1 |
| **S4** | AddRoutineView 保存后列表不自动刷新 | 用户可能认为保存失败 | P1 |
| **S5** | RoutineDetailView 保存后列表不自动刷新 | 同 S4 | P1 |

### V4.0 优化目标

```
┌─────────────────────────────────────────────────────────────────────┐
│                         FreshFace V4.0 目标                          │
```

## 问题分析

### S1: RoutineStepEditorView availableProducts 为空

**现有代码分析**:
```swift
// RoutineDetailView.swift - 第30-35行
.sheet(isPresented: $showAddStep) {
    RoutineStepEditorView(
        step: nil,
        availableProducts: []  // ⚠️ 空数组 - 问题所在
```

**问题**:
- RoutineStepEditorView 需要 availableProducts 让用户选择已有产品

**解决方案**:
- 在 RoutineDetailView 中获取产品列表并传入
- 需要从 CloudProductRepository 获取

### S3-S5: 数据刷新问题

**现有代码分析**:
```swift
// ProductListView.swift - 第67行

    viewModel.loadProducts()
}
```

**问题**:
- NavigationStack 中嵌套 TabView 时，.refreshable 可能不会生效
- 用户从 ProductDetailView 返回后列表不会自动刷新
- RoutineListView 也有类似问题

**解决方案**:
- 使用 @Observable 配合 .observationCalendar()
- 或在 view.onAppear 中主动调用 loadProducts()
- 确保数据变更后及时刷新

```swift
struct RoutineDetailView: View {
    @State var routine: Routine
    @State private var showAddStep = false
    @State private var editingStep: RoutineStep?
    @State private var isLoading = false
    @State private var errorMessage: String?
    @State private var availableProducts: [Product] = []  // ✅ 添加产品列表状态

    private let repository = CloudRoutineRepository()

    var body: some View {
        List {
            infoSection
            stepsSection
            addStepButton
        }
        // ...
        .sheet(isPresented: $showAddStep) {
            RoutineStepEditorView(
                step: nil,
                availableProducts: availableProducts  // ✅ 传入真实列表
            ) { newStep in
                addStep(newStep)
            }
        }
        .sheet(item: $editingStep) { step in
            RoutineStepEditorView(
                step: step,

            availableProducts = try await CloudProductRepository().fetchAll()

        }
    }
```

```swift
// AddProductViewModel.swift 修改
class AddProductViewModel: ObservableObject {
    // ... existing properties ...

    func lookupProduct(barcode: String) async -> ProductLookupResult {
        // ✅ 实现 ProductLookupService 调用
        do {
            if let product = try await ProductLookupService.shared.lookupByBarcode(barcode) {
                return .found(product)
            } else {
                return .notFound
            }
        } catch {
            return .error(error.localizedDescription)
        }
    }

    func applyLookupResult(_ result: ProductLookupResult) {
        switch result {
        case .found(let product):
            // ✅ 自动填充产品信息
            name = product.name
            brand = product.brand ?? ""
            category = product.category
            productType = product.productType
        case .notFound, .error:
            // ⚠️ 只保留条码，其他信息需要用户手动输入
            break
        }
    }
}

// ProductLookupService.swift 完善
class ProductLookupService {
    static let shared = ProductLookupService()

    func lookupByBarcode(_ barcode: String) async throws -> Product? {
        // 实现 OpenFoodFacts API 调用
        // 或使用 Makeup API

        // 示例：OpenFoodFacts API
        let urlString = "https://world.openfoodfacts.org/api/v0/product/\(barcode).json"
        guard let url = URL(string: urlString) else { return nil }

        let (data, _) = try await URLSession.shared.data(from: url)
        let response = try JSONDecoder().decode(OpenFoodFactsResponse.self, from: data)

        guard response.status == 1, let product = response.product else {
            return nil
        }

        return Product(
            name: product.productName ?? "Unknown Product",
            brand: product.brands,
            category: inferCategory(from: product.categories),
            productType: inferProductType(from: product.categories),
            barcode: barcode
        )
    }

    private func inferCategory(from categories: String?) -> ProductCategory {
        guard let categories = categories?.lowercased() else { return .other }
        if categories.contains("skincare") || categories.contains("face") {
            return .skincare
        } else if categories.contains("makeup") || categories.contains("cosmetic") {
            return .makeup
        } else if categories.contains("hair") {
            return .haircare
        } else if categories.contains("fragrance") || categories.contains("perfume") {
            return .fragrance
        }
        return .other
    }

    private func inferProductType(from categories: String?) -> ProductType? {
        // 根据 categories 推断具体产品类型
        return nil
    }
}
```

```swift
// AddRoutineView.swift 修改
struct AddRoutineView: View {
    @Environment(\.dismiss) var dismiss
    @State private var name = ""
    @State private var routineType: RoutineType = .morning
    @State private var isSaving = false
    @State private var errorMessage: String?

    let onSave: (Routine) -> Void

    var body: some View {
        NavigationStack {
            Form {
                Section(header: Text("Routine Info")) {
                    TextField("Routine Name", text: $name)

                    Picker("Type", selection: $routineType) {
                        ForEach(RoutineType.allCases, id: \.self) { type in
                            Text(type.displayName).tag(type)
                        }
                    }
                }
            }
            .navigationTitle("New Routine")
            .navigationBarTitleDisplayMode(.inline)
            .toolbar {
                ToolbarItem(placement: .cancellationAction) {
                    Button("Cancel") { dismiss() }
                }
                ToolbarItem(placement: .confirmationAction) {
                    Button("Save") {
                        saveRoutine()
                    }
                    .disabled(name.isEmpty || isSaving)
                }
            }
        }
    }

    private func saveRoutine() {
        isSaving = true

        let newRoutine = Routine(
            name: name,
            routineType: routineType,
            isActive: true
        )

        Task {
            do {
                let repository = CloudRoutineRepository()
                let created = try await repository.create(newRoutine)
                await MainActor.run {
                    onSave(created)  // ✅ 传入创建后的完整对象
                    isSaving = false
                    dismiss()
                }
            } catch {
                await MainActor.run {
                    errorMessage = "Failed to save routine: \(error.localizedDescription)"
                    isSaving = false
                }
            }
        }
    }
}
```

```swift
// RoutineDetailView.swift 修改
struct RoutineDetailView: View {
    @State var routine: Routine
    @State private var showAddStep = false
    @State private var editingStep: RoutineStep?
    @State private var isLoading = false
    @State private var errorMessage: String?
    @State private var isSaving = false

    private let repository = CloudRoutineRepository()

    var body: some View {
        // ...
    }

    private func addStep(_ step: RoutineStep) {
        Task {
            var updatedRoutine = routine
            updatedRoutine.steps.append(step)
            await saveRoutine(updatedRoutine)
        }
    }

    private func updateStep(_ step: RoutineStep) {
        Task {
            var updatedRoutine = routine
            if let index = updatedRoutine.steps.firstIndex(where: { $0.id == step.id }) {
                updatedRoutine.steps[index] = step
            }
            await saveRoutine(updatedRoutine)
        }
    }

    private func deleteStep(_ step: RoutineStep) {
        Task {
            var updatedRoutine = routine
            updatedRoutine.steps.removeAll { $0.id == step.id }
            await saveRoutine(updatedRoutine)
        }
    }

    private func saveRoutine(_ routine: Routine) async {
        isSaving = true
        do {
            let updated = try await repository.update(routine)
            await MainActor.run {
                self.routine = updated  // ✅ 更新本地状态
                isSaving = false
            }
        } catch {
            await MainActor.run {
                errorMessage = "Failed to save: \(error.localizedDescription)"
                isSaving = false
            }
        }
    }
}
```

## 设计原则与约束

### 1. 功能模块原则
- 每个修复作为独立模块实现
- 文件命名清晰，位置正确
- 遵循 Single Responsibility Principle

### 2. 代码复用原则
- 优先复用现有组件
- ProductLookupService 已存在，需集成而非重写
- Repository 模式已建立，直接使用

### 3. 代码清理原则
- 替换时标记废弃 → 验证 → 删除
- 不破坏现有功能

### 4. 风格一致性原则
- 所有 UI 文本保持英文
- 遵循 SwiftUI iOS 17+ 最佳实践
- 使用现有 Color(hex:) 扩展保持品牌一致性

### 5. 继承与兼容性原则
- 充分利用现有 Repository 架构
- 不硬编码，直接使用 CloudProductRepository 等
- SettingsView 中的 NotificationSettingsDetailView 已集成，需保持兼容
