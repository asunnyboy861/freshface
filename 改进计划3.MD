| 问题编号 | 问题描述 | 影响 | 优先级 |
|---------|---------|------|--------|
| **S1** | Settings 中 "Show Onboarding" 按钮未实现 sheet | 用户无法重新查看引导 | P0 |

// ProductDetailView.swift 修改
struct ProductDetailView: View {
    @StateObject private var viewModel: ProductDetailViewModel
    @State private var showLogUsageSheet = false
    @State private var usageLogs: [UsageLog] = []

    private let usageLogRepository = CloudUsageLogRepository()

    var body: some View {
        ScrollView {
            VStack(spacing: 20) {
                productImageView
                productInfoSection
                expiryStatusSection
                completionPredictionSection
                usageSection  // 新增
                actionsSection
            }
            .padding()
        }
        .sheet(isPresented: $showLogUsageSheet) {
            LogUsageView(product: viewModel.product) { log in
                logUsage(log)
            }
        }
    }

    private var usageSection: some View {
        VStack(alignment: .leading, spacing: 12) {
            HStack {
                Text("Usage History")
                    .font(.headline)

                Spacer()

                Button(action: { showLogUsageSheet = true }) {
                    Label("Log Usage", systemImage: "plus.circle.fill")
                        .font(.subheadline)
                }
            }

            if usageLogs.isEmpty {
                Text("No usage recorded yet")
                    .font(.caption)
                    .foregroundColor(.secondary)
            } else {
                ForEach(usageLogs.prefix(5)) { log in
                    HStack {
                        Image(systemName: "checkmark.circle.fill")
                            .foregroundColor(.green)
                        Text(log.date.formatted(date: .abbreviated, time: .shortened))
                            .font(.caption)
                        Spacer()
                    }
                }
            }
        }
        .padding()
        .background(Color(.systemGray6))
        .cornerRadius(12)
    }

    private func logUsage(_ log: UsageLog) {
        Task {
            do {
                _ = try await usageLogRepository.create(log)
                await loadUsageLogs()
            } catch {
                print("Error logging usage: \(error)")
            }
        }
    }

    private func loadUsageLogs() async {
        do {
            usageLogs = try await usageLogRepository.fetch(for: viewModel.product.id)
        } catch {
            print("Error loading usage logs: \(error)")
        }
    }
}

// LogUsageView.swift
struct LogUsageView: View {
    let product: Product
    let onSave: (UsageLog) -> Void

    @Environment(\.dismiss) var dismiss
    @State private var notes = ""
    @State private var skinCondition = ""

    var body: some View {
        NavigationStack {
            Form {
                Section(header: Text("Log Usage")) {
                    TextField("Notes (optional)", text: $notes, axis: .vertical)
                        .lineLimit(3...6)

                    TextField("Skin Condition (optional)", text: $skinCondition, axis: .vertical)
                        .lineLimit(2...4)
                }

                Section(header: Text("Summary")) {
                    HStack {
                        Text("Product")
                        Spacer()
                        Text(product.name)
                            .foregroundColor(.secondary)
                    }

                    HStack {
                        Text("Time")
                        Spacer()
                        Text(Date().formatted(date: .abbreviated, time: .shortened))
                            .foregroundColor(.secondary)
                    }
                }
            }
            .navigationTitle("Log Usage")
            .navigationBarTitleDisplayMode(.inline)
            .toolbar {
                ToolbarItem(placement: .cancellationAction) {
                    Button("Cancel") { dismiss() }
                }
                ToolbarItem(placement: .confirmationAction) {
                    Button("Save") {
                        let log = UsageLog(
                            productId: product.id,
                            notes: notes.isEmpty ? nil : notes,
                            skinCondition: skinCondition.isEmpty ? nil : skinCondition
                        )
                        onSave(log)
                        dismiss()
                    }
                }
            }
        }
    }
}

// AddProductView.swift 修改
struct AddProductView: View {
    @StateObject private var viewModel: AddProductViewModel
    @Environment(\.dismiss) var dismiss

    init(repository: ProductRepositoryProtocol) {
        _viewModel = StateObject(wrappedValue: AddProductViewModel(repository: repository))
    }

    var body: some View {
        NavigationStack {
            Form {
                // ... sections
            }
            .toolbar {
                ToolbarItem(placement: .confirmationAction) {
                    Button("Save") {
                        if viewModel.saveProduct() != nil {
                            viewModel.resetForm()  // ✅ 重置表单
                            dismiss()  // ✅ 关闭页面
                        }
                    }
                    .disabled(!viewModel.isValidForm || viewModel.isSaving)
                }
            }
        }
    }
}

// AnalyticsChartsView.swift
import SwiftUI
import Charts

struct AnalyticsChartsView: View {
    let products: [Product]

    var body: some View {
        ScrollView {
            VStack(spacing: 24) {
                expiryTrendChart
                categoryPieChart
            }
            .padding()
        }
    }

    private var expiryTrendChart: some View {
        VStack(alignment: .leading, spacing: 12) {
            Text("Expiry Trend")
                .font(.headline)

            Chart {
                ForEach(aggregatedData, id: \.month) { item in
                    BarMark(
                        x: .value("Month", item.month),
                        y: .value("Count", item.count)
                    )
                    .foregroundStyle(Color(hex: "FF6B9D").gradient)
                }
            }
            .frame(height: 200)
        }
        .padding()
        .background(Color(.systemBackground))
        .cornerRadius(12)
    }

    private var categoryPieChart: some View {
        VStack(alignment: .leading, spacing: 12) {
            Text("Category Distribution")
                .font(.headline)

            Chart(products, id: \.id) { product in
                SectorMark(
                    angle: .value("Count", 1),
                    innerRadius: .ratio(0.5),
                    angularInset: 1.5
                )
                .cornerRadius(4)
                .foregroundStyle(by: .value("Category", product.category.displayName))
            }
            .frame(height: 200)
        }
        .padding()
        .background(Color(.systemBackground))
        .cornerRadius(12)
    }
}

// ShareProductView.swift
struct ShareProductView: View {
    let product: Product
    @State private var shareImage: UIImage?
    @State private var showShareSheet = false

    var body: some View {
        Button(action: generateShareImage) {
            Label("Share Product", systemImage: "square.and.arrow.up")
        }
        .sheet(isPresented: $showShareSheet) {
            if let image = shareImage {
                ShareSheet(items: [image])
            }
        }
    }

    private func generateShareImage() {
        let renderer = ImageRenderer(content: productCardView)
        renderer.scale = UIScreen.main.scale
        shareImage = renderer.uiImage
        showShareSheet = true
    }

    private var productCardView: some View {
        VStack(spacing: 16) {
            if let imageData = product.imageData,
               let uiImage = UIImage(data: imageData) {
                Image(uiImage: uiImage)
                    .resizable()
                    .scaledToFit()
                    .frame(width: 200, height: 200)
                    .clipShape(RoundedRectangle(cornerRadius: 12))
            }

            Text(product.name)
                .font(.title2.bold())

            if let brand = product.brand {
                Text(brand)
                    .font(.subheadline)
                    .foregroundColor(.secondary)
            }

            HStack(spacing: 20) {
                Label("\(product.daysRemaining ?? 0) days", systemImage: "clock")
                Label(product.category.displayName, systemImage: product.category.icon)
            }
            .font(.caption)
        }
        .padding(32)
        .background(Color(.systemBackground))
    }
}

struct ShareSheet: UIViewControllerRepresentable {
    let items: [Any]

    func makeUIViewController(context: Context) -> UIActivityViewController {
        UIActivityViewController(activityItems: items, applicationActivities: nil)
    }

    func updateUIViewController(_ uiViewController: UIActivityViewController, context: Context) {}
}

│  • ViewModel 与 View 同名放置                                       │
└─────────────────────────────────────────────────────────────────────┘
```

### 2. 代码复用原则

```
┌─────────────────────────────────────────────────────────────────────┐
│                           代码复用规范                                │
├─────────────────────────────────────────────────────────────────────┤
│  • 相似组件: 抽象为通用组件 (如 LoadingStateView)                   │
│  • 复用条件: 出现3次相同代码时抽取                                    │
│  • 继承关系: 优先使用协议和泛型，避免硬编码                            │
└─────────────────────────────────────────────────────────────────────┘
```

### 3. 重构清理原则

```
┌─────────────────────────────────────────────────────────────────────┐
│                           重构清理规范                                │
├─────────────────────────────────────────────────────────────────────┤
│  1. 标记废弃: 使用 @available(*, deprecated)                         │
│  2. 验证影响: 运行所有相关测试                                        │
│  3. 删除冗余: 确认无引用后删除                                        │
│  4. 提交记录: 注明清理范围                                            │
