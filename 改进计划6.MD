# FreshFace 改进计划 6 - 成分分析与产品推荐系统

## 一、需求描述

### 1.1 问题背景
基于深度分析报告，FreshFace 在美国市场的核心竞争力存在明显短板：
- **成分分析功能缺失**：美国护肤用户高度关注产品成分安全性（Clean Beauty趋势）
- **无产品推荐能力**：竞品（ThinkDirty, Yuka）均提供成分评分和替代品推荐
- **数据价值未挖掘**：已有产品数据仅用于过期追踪，未产生洞察价值

### 1.2 竞品对标分析
| 功能 | ThinkDirty | Yuka | FreshFace现状 | 目标 |
|------|------------|------|---------------|------|
| 成分扫描 | ✅ 条码/拍照 | ✅ 条码 | ✅ 条码 | 持平 |
| 成分数据库 | ✅ 100万+ | ✅ 120万+ | ❌ 无 | **需添加** |
| 危险成分标记 | ✅ 红色警告 | ✅ 评分0-100 | ❌ 无 | **需添加** |
| 产品推荐 | ✅ 清洁替代品 | ✅  healthier选项 | ❌ 无 | **需添加** |
| 护肤流程 | ❌ 不支持 | ❌ 不支持 | ✅ 支持 | **优势** |

### 1.3 用户价值主张
**For美国用户**: "Not just track expiry, but understand what you put on your skin"
- 扫描产品 → 查看成分安全评分 → 发现有害成分 → 获得清洁替代品推荐

### 1.4 功能范围（MVP）
1. **成分数据库** - 内置常见护肤成分安全评级
2. **产品成分分析** - 手动输入/扫描后解析成分列表
3. **安全评分系统** - 0-100分制，颜色编码
4. **有害成分高亮** - 标记parabens, sulfates等争议成分
5. **清洁产品推荐** - 基于类别推荐高评分替代品

```swift
class IngredientDatabase {
    static let shared = IngredientDatabase()
    
    private var ingredients: [Ingredient] = []
    private var nameIndex: [String: Ingredient] = [:]      // 名称索引
    private var aliasIndex: [String: Ingredient] = [:]     // 别名索引
    
    private init() {
        loadDefaultIngredients()
        buildIndexes()
    }
    
    // MARK: - 公共API
    
    /// 根据名称查找成分
    func findIngredient(byName name: String) -> Ingredient? {
        let normalizedName = normalize(name)
        return nameIndex[normalizedName] ?? aliasIndex[normalizedName]
    }
    
    /// 搜索成分（支持部分匹配）
    func searchIngredients(query: String) -> [Ingredient] {
        let normalizedQuery = normalize(query)
        return ingredients.filter { ingredient in
            normalize(ingredient.name).contains(normalizedQuery) ||
            ingredient.aliases.contains { normalize($0).contains(normalizedQuery) }
        }
    }
    
    /// 获取所有成分
    func getAllIngredients() -> [Ingredient] {
        return ingredients
    }
    
    /// 按安全评级筛选
    func getIngredients(byRating rating: SafetyRating) -> [Ingredient] {
        return ingredients.filter { $0.safetyRating == rating }
    }
    
    /// 按关注问题筛选
    func getIngredients(byConcern concern: SafetyConcern) -> [Ingredient] {
        return ingredients.filter { $0.concerns.contains(concern) }
```

#### 2.2.4 成分解析服务 (Services/IngredientAnalysis/IngredientParserService.swift)
```swift
import Foundation

/// 成分列表解析服务
/// 将产品成分字符串解析为结构化数据
class IngredientParserService {
    static let shared = IngredientParserService()
    
    private let database = IngredientDatabase.shared
    
    private init() {}
    
    // MARK: - 公共API
    
    /// 解析成分列表字符串
    /// - Parameter ingredientList: 逗号分隔的成分列表
    /// - Returns: 解析结果，包含识别到的成分和未识别的成分
    func parse(ingredientList: String) -> IngredientParseResult {
        let rawIngredients = splitIngredients(ingredientList)
        var recognized: [ParsedIngredient] = []
        var unrecognized: [String] = []
        
        for raw in rawIngredients {
            if let ingredient = database.findIngredient(byName: raw) {
                recognized.append(ParsedIngredient(
                    rawText: raw,
                    ingredient: ingredient,
                    confidence: .high
                ))
            } else {
                // 尝试模糊匹配
                let searchResults = database.searchIngredients(query: raw)
                if let bestMatch = searchResults.first {
                    recognized.append(ParsedIngredient(
                        rawText: raw,
                        ingredient: bestMatch,
                        confidence: .medium
                    ))
                } else {
                    unrecognized.append(raw)
                }
            }
        }
        
        return IngredientParseResult(
            recognizedIngredients: recognized,
            unrecognizedIngredients: unrecognized,
            totalCount: rawIngredients.count
        )
    }
    
    /// 从条码扫描结果解析成分
    func parseFromBarcodeResult(_ result: ProductLookupResult) -> IngredientParseResult? {
        guard let ingredients = result.ingredients else { return nil }
        return parse(ingredientList: ingredients)
    }
    
    // MARK: - 私有方法
    
    private func splitIngredients(_ text: String) -> [String] {
        // 支持多种分隔符：逗号、分号、换行
        let separators = CharacterSet(charactersIn: ",;|\n")
        return text
            .components(separatedBy: separators)
            .map { $0.trimmingCharacters(in: .whitespaces) }
            .filter { !$0.isEmpty }
    }
}

// MARK: - 解析结果模型

struct IngredientParseResult {
    let recognizedIngredients: [ParsedIngredient]
    let unrecognizedIngredients: [String]
    let totalCount: Int
    
    var recognizedCount: Int { recognizedIngredients.count }
    var recognitionRate: Double {
        guard totalCount > 0 else { return 0 }
        return Double(recognizedCount) / Double(totalCount)
    }
}

struct ParsedIngredient {
    let rawText: String           // 原始文本
    let ingredient: Ingredient    // 匹配到的成分
    let confidence: MatchConfidence // 匹配置信度
    
    enum MatchConfidence {
        case high      // 精确匹配
        case medium    // 模糊匹配
        case low       // 可能匹配
    }
}
```

- [ ] 返回识别率和未识别成分列表
- [ ] 实现模糊匹配机制
- [ ] 与现有 ProductLookupService 集成

#### 2.2.5 安全评分计算 (Services/IngredientAnalysis/SafetyRatingCalculator.swift)
```swift
import Foundation

/// 产品安全评分计算器
/// 基于成分分析计算整体安全评分
class SafetyRatingCalculator {
    static let shared = SafetyRatingCalculator()
    
    private init() {}
    
    // MARK: - 公共API
    
    /// 计算产品整体安全评分
    func calculateOverallScore(from parseResult: IngredientParseResult) -> ProductSafetyScore {
        let ingredients = parseResult.recognizedIngredients
        
        guard !ingredients.isEmpty else {
            return ProductSafetyScore(
                overallScore: 0,
                rating: .unknown,
                ingredientBreakdown: [],
                topConcerns: [],
                recommendation: "Unable to analyze ingredients"
            )
        }
        
        // 计算加权平均分
        let totalScore = ingredients.reduce(0) { sum, parsed in
            sum + parsed.ingredient.safetyRating.rawValue
        }
        let averageScore = totalScore / ingredients.count
        
        // 确定整体评级
        let overallRating = SafetyRating(
            rawValue: (averageScore / 20) * 20  // 归一化到枚举值
        ) ?? .unknown
        
        // 提取各成分评分详情
        let breakdown = ingredients.map { parsed in
            IngredientScoreDetail(
                name: parsed.ingredient.name,
                rawText: parsed.rawText,
                score: parsed.ingredient.safetyRating.rawValue,
                rating: parsed.ingredient.safetyRating,
                concerns: parsed.ingredient.concerns
            )
        }.sorted { $0.score < $1.score }  // 按风险排序
        
        // 提取主要关注问题
        let allConcerns = ingredients.flatMap { $0.ingredient.concerns }
        let concernCounts = Dictionary(grouping: allConcerns) { $0 }
            .mapValues { $0.count }
        let topConcerns = concernCounts.sorted { $0.value > $1.value }
            .prefix(3)
            .map { $0.key }
        
        // 生成建议
        let recommendation = generateRecommendation(
            overallRating: overallRating,
            topConcerns: Array(topConcerns),
            unrecognizedCount: parseResult.unrecognizedIngredients.count
        )
        
        return ProductSafetyScore(
            overallScore: averageScore,
            rating: overallRating,
            ingredientBreakdown: breakdown,
            topConcerns: Array(topConcerns),
            recommendation: recommendation
        )
    }
    
    // MARK: - 私有方法
    
    private func generateRecommendation(
        overallRating: SafetyRating,
        topConcerns: [SafetyConcern],
        unrecognizedCount: Int
    ) -> String {
        var parts: [String] = []
        
        switch overallRating {
        case .verySafe, .safe:
            parts.append("This product contains mostly safe ingredients.")
        case .moderate:
            parts.append("This product has some ingredients of moderate concern.")
        case .concerning:
            parts.append("This product contains several ingredients of concern.")
        case .unsafe:
            parts.append("This product contains ingredients we recommend avoiding.")
        case .unknown:
            parts.append("Unable to fully analyze this product's safety.")
        }
        
        if !topConcerns.isEmpty {
            let concernNames = topConcerns.map { $0.rawValue }.joined(separator: ", ")
            parts.append("Main concerns: \(concernNames).")
        }
        
        if unrecognizedCount > 0 {
            parts.append("\(unrecognizedCount) ingredient(s) not recognized in our database.")
        }
        
        return parts.joined(separator: " ")
```

### 3.3 产品推荐算法设计

**推荐逻辑（补充）：**

```swift
class ProductRecommendationsViewModel: ObservableObject {
    @Published var recommendations: [ProductRecommendation] = []
    
    func loadRecommendations(for category: ProductCategory, excluding excludedId: UUID?) {
        // 1. 获取用户库中该类别的所有产品
        let userProducts = LocalProductRepository().fetchByCategory(category)
        
        // 2. 筛选有成分分析且评分高的产品（>=80分）
        let highRatedProducts = userProducts
            .filter { $0.ingredients != nil }
            .compactMap { product -> ProductRecommendation? in
                guard let parseResult = IngredientParserService.shared.parseFromProduct(product),
                      let score = SafetyRatingCalculator.shared.calculateScore(parseResult),
                      score.overallScore >= 80 else { return nil }
                
                return ProductRecommendation(
                    product: product,
                    safetyScore: score,
                    recommendationReason: "High safety rating in your library"
                )
            }
        
        // 3. 按评分排序
        recommendations = highRatedProducts.sorted { 
            $0.safetyScore.overallScore > $1.safetyScore.overallScore 
        }
    }
}
```

## 五、成功指标

| 指标 | 目标值 | 测量方式 |
|------|--------|----------|
| 成分识别率 | >= 70% | 测试常见产品成分列表 |
| 评分计算准确性 | 100% | 单元测试覆盖 |
| 用户添加成分转化率 | >= 30% | 分析无成分产品的用户行为 |
| Clean产品推荐点击率 | >= 20% | 推荐卡片点击统计 |

## 六、风险评估

| 风险 | 影响 | 缓解措施 |
|------|------|----------|
| 成分数据库不完整 | 中 | 从最常见成分开始，逐步扩展 |
| 评分算法争议 | 中 | 参考EWG、ThinkDirty等权威来源 |
| 性能问题（大数据量） | 低 | 使用索引优化查询 |
