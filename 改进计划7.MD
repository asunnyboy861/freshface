# 改进计划7.MD - UI/UX优化与美国市场适配

## 一、需求概述

### 1.1 背景
基于对美国市场竞争产品（ThinkDirty、Yuka、Skincare Routine）的深度分析，FreshFace需要优化以下方面以更好地适配美国市场用户习惯：

| 问题 | 当前状态 | 目标状态 | 优先级 |
|------|----------|----------|--------|
| Dark Mode | 仅在Settings切换，未全局应用 | 全局自动跟随系统设置 | **高** |
| 肤质档案 | 无 | 用户首次使用时选择肤质 | **高** |
| 成分数据库 | 35+成分 | 100+常见护肤成分 | **中** |

### 1.2 目标
1. 实现真正的Dark Mode支持，自动跟随iOS系统设置
2. 添加肤质档案功能，支持个性化推荐
3. 扩展成分数据库至100+常见护肤成分

## 二、详细设计方案

### 2.1 Dark Mode 全局应用

#### 需求描述
当前Dark Mode仅在Settings中通过`overrideUserInterfaceStyle`强制切换，这种方式：
- ❌ 不符合Apple HIG推荐的`.preferredColorScheme()`方式
- ❌ 可能导致某些视图无法正确响应
- ❌ 性能较差，需要手动应用到每个window

#### 技术方案
修改`freshfaceApp.swift`，使用SwiftUI原生的`@AppStorage`配合`.preferredColorScheme()`：

```swift
// freshfaceApp.swift
import SwiftUI

@main
struct FreshFaceApp: App {
    @AppStorage("darkModeEnabled") private var darkModeEnabled = false
    @AppStorage("useSystemAppearance") private var useSystemAppearance = true
    @StateObject private var appCoordinator = AppCoordinator()
    @AppStorage("hasCompletedOnboarding") private var hasCompletedOnboarding = false

    var body: some Scene {
        WindowGroup {
            if hasCompletedOnboarding {
                AppRootView()
                    .environmentObject(appCoordinator)
                    .preferredColorScheme(colorScheme)
            } else {
                OnboardingView(hasCompleted: $hasCompletedOnboarding)
                    .preferredColorScheme(colorScheme)
            }
        }
    }

    private var colorScheme: ColorScheme? {
        if useSystemAppearance {
            return nil
        }
        return darkModeEnabled ? .dark : .light
    }
}
```

#### Settings界面更新
```swift
// SettingsView.swift - appearanceSection (修改现有代码)
private var appearanceSection: some View {
    Section(header: Text("Appearance")) {
        Toggle("Use System Appearance", isOn: $useSystemAppearance)

            Toggle("Dark Mode", isOn: $darkModeEnabled)
        }
    }
}
```

**注意**：`applyAppearance()`函数可以删除，因为不再需要手动设置window样式。

#### Storage服务

```swift
// SkinProfileStorage.swift - Data/Storages/
class SkinProfileStorage {
    static let shared = SkinProfileStorage()

    private let userDefaults = UserDefaults.standard
    private let storageKey = "skinProfile"

    private init() {}

    func save(_ profile: SkinProfile) {
        if let encoded = try? JSONEncoder().encode(profile) {
            userDefaults.set(encoded, forKey: storageKey)
        }
    }

    func load() -> SkinProfile? {
        guard let data = userDefaults.data(forKey: storageKey),
              let profile = try? JSONDecoder().decode(SkinProfile.self, from: data) else {
            return nil
        }
        return profile
    }

    func clear() {
        userDefaults.removeObject(forKey: storageKey)
    }
}
```

                .font(.subheadline)
                .foregroundColor(.secondary)
                .multilineTextAlignment(.center)
        }
    }

    private var skinTypeSection: some View {
        VStack(alignment: .leading, spacing: 16) {
            Text("What is your skin type?")
                .font(.headline)

            LazyVGrid(columns: [GridItem(.flexible()), GridItem(.flexible())], spacing: 12) {
                ForEach(SkinType.allCases) { type in
                    SkinTypeCard(
                        skinType: type,
                        isSelected: selectedSkinType == type
                    )
                    .onTapGesture {
                        withAnimation(.easeInOut(duration: 0.2)) {
                            selectedSkinType = type
                        }
                    }
                }
            }
        }
    }

    private var concernsSection: some View {
        VStack(alignment: .leading, spacing: 16) {
            Text("Any skin concerns?")
                .font(.headline)
            Text("Select all that apply (optional)")
                .font(.caption)
                .foregroundColor(.secondary)

            LazyVGrid(columns: [GridItem(.flexible()), GridItem(.flexible())], spacing: 12) {
                ForEach(SkinConcern.allCases) { concern in
                    ConcernChip(
                        concern: concern,
                        isSelected: selectedConcerns.contains(concern)
                    )
                    .onTapGesture {
                        if selectedConcerns.contains(concern) {
                            selectedConcerns.remove(concern)
                        } else {
                            selectedConcerns.insert(concern)
                        }
                    }
                }
            }
        }
    }

    private var saveButton: some View {
        Button(action: saveProfile) {
            Text("Continue")
                .font(.headline)
                .foregroundColor(.white)
                .frame(maxWidth: .infinity)
                .padding()
                .background(selectedSkinType != nil ? Color(hex: "FF6B9D") : Color.gray)
                .cornerRadius(12)
        }
        .disabled(selectedSkinType == nil)
    }

    private func saveProfile() {
        guard let skinType = selectedSkinType else { return }

        let profile = SkinProfile(
            skinType: skinType,
            primaryConcerns: Array(selectedConcerns)
        )

        SkinProfileStorage.shared.save(profile)
        hasCompleted = true
    }
}

        .padding(.horizontal, 12)
        .padding(.vertical, 8)

    }
}

}

func testNewIngredients() {
    let database = IngredientDatabase.shared
    let ingredient = database.findIngredient(byName: "Alpha Arbutin")
    XCTAssertNotNil(ingredient, "Alpha Arbutin should be in database")
    XCTAssertEqual(ingredient?.safetyRating, .safe)
}

func testCategoryCoverage() {
    let database = IngredientDatabase.shared
    let categories: [IngredientCategory] = [.humectant, .emollient, .surfactant, .preservative, .antioxidant]
    for category in categories {
        let count = database.getAllIngredients().filter { $0.categories.contains(category) }.count
        XCTAssertGreaterThan(count, 5, "Each category should have at least 5 ingredients")
    }
}

## 六、风险与注意事项

1. **Dark Mode性能**：确保所有自定义颜色都支持light/dark变体
2. **数据迁移**：SkinProfileStorage需兼容首次安装场景
3. **成分数据准确性**：风险评级需参考EWG Skin Deep数据库
4. **iOS版本兼容性**：preferredColorScheme从iOS 13+可用，当前项目支持iOS 17+无需担心
5. **枚举值兼容性**：使用现有的IngredientCategory枚举值（.emollient代替.occlusive）
